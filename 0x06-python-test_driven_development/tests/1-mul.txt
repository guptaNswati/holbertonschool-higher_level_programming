#/usr/bin/python3
"""
This is the "Multiplication" module.

The Multiplication module supplies a simple function, mul()
    to multiply 2 Python variables. For example,

>>> mul(4, 5)
20
"""

def mul(a, b):
    """
    Return the multiplication of 2 arguements.
    >>> [mul(2, 3)]
    [6]
    >>> [mul(100.3, -2)]
    [-200]
    >>> [mul(100.3, "School")]
    [b must be an integer]
    >>> [mul("Holberton", "School")]
    [a and b must be integers]
    """
    return (a * b)
